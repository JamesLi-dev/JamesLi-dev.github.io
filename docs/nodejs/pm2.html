PM2

官网地址​

  1. 社区中使用最广泛的 Nodejs 进程管理工具，支持进程负载均衡

  2. 当一个进程宕机，pm2 会自动重启该服务，保证 Node.js 应用的高可用性

  3. 提供一部分 APM 能力，如进程监控等

使用​

安装​

npm install pm2 -g

常用核心命令​

# 1. 进入项目启动
pm2 start xxx.config.js

# 2. 查看运行目录
pm2 ls

# 3. 停止 stop
pm2 stop [name或id]
# 停止所有
pm2 stop all

# 4. 查看某个进程的信息
pm2 describe 0

# 5. 查看日志
pm2 log

# 6. 删除进程
pm2 delete [xxx | all]
# kill 杀掉进程
pm2 kill

# 7. 重启
pm2 restart xxx.js
# 或者重启已存在的进程
pm2 reload [xxx| all]

配置文件​

1. 初始化​

# 在项目根目录下运行配置初始化命令
pm2 init simple

2. 初始 pm2 配置文件 ecosystem.config.js​

//
module.exports = {
  apps: [
    {
      name: "app1",
      script: "./app.js",
    },
  ],
};

3. 启动​

pm2 start ./ecosystem.con

推荐配置​

只需修改 name 和 script 路径即可

module.exports = {
  apps: [
    {
      /**
       * 项目主配置
       */
      // 应用名称
      name: "test-xxx",
      // 项目启动的脚本
      script: "./app.js",
      // 项目运行的根目录
      cwd: "./",
      // 监控变化的目录，一旦变化，自动重启
      watch: ["controllers", "dataBase", "utils", "view"],
      // 从监控目录中排除
      ignore_watch: ["node_modules", "logs"],
      watch_options: {
        followSymlinks: false,
      },

      /**
       * Control flow 配置
       */
      // 项目最小健康时间：我们这里认为该项目启动的 24h 内一定都是正常运行的，否则将产生 error 报错
      min_uptime: "24h",
      // 项目最大内存：当内存占用超过这个 limit 时将自动重启
      max_memory_restart: "500M",
      // 自动重启次数
      max_restarts: 10,
      // 每次尝试重启的间隔时间（毫秒）
      restart_delay: 4000,

      /**
       * Log file 配置
       */
      // 输出日志时的时间日期 prefix，每行日志前都会带上
      log_date_format: "YYYY-MM-DD HH:mm Z",
      // 程序运行控制台的正常输出保存位置
      out_file: "./log/out.log",
      // 程序运行错误日志保存位置
      error_file: "./log/error.log",
      // 是否将多个 pid 的进程日志合并，我们不希望日志还要分开看，所以合并在一个文件查看即可
      combine_logs: false,
    },
    // 多个pm2进程，如不需要，以下代码可删除
    {
      /**
       * 项目主配置
       */
      // 应用名称
      name: "syncTimer",
      // 项目启动的脚本
      script: "./utils/syncTimer.js",
      // 项目运行的根目录
      cwd: "./",
      // 监控变化的目录，一旦变化，自动重启
      watch: ["controllers", "dataBase", "utils", "view"],
      // 从监控目录中排除
      ignore_watch: ["node_modules", "logs"],
      watch_options: {
        followSymlinks: false,
      },

      /**
       * Control flow 配置
       */
      // 项目最小健康时间：我们这里认为该项目启动的 24h 内一定都是正常运行的，否则将产生 error 报错
      min_uptime: "24h",
      // 项目最大内存：当内存占用超过这个 limit 时将自动重启
      max_memory_restart: "500M",
      // 自动重启次数
      max_restarts: 10,
      // 每次尝试重启的间隔时间（毫秒）
      restart_delay: 4000,

      /**
       * Log file 配置
       */
      // 输出日志时的时间日期 prefix，每行日志前都会带上
      log_date_format: "YYYY-MM-DD HH:mm Z",
      // 程序运行控制台的正常输出保存位置
      out_file: "./log/syncTimerOut.log",
      // 程序运行错误日志保存位置
      error_file: "./log/syncTimerError.log",
      // 是否将多个 pid 的进程日志合并，我们不希望日志还要分开看，所以合并在一个文件查看即可

    },
  ],
};
